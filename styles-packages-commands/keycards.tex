%
% this command evalutes to the value of the section label, depending on the section counters
% intermediate counters are needed, because \thesubsection and \thesubsubsection dont evalute to just the number of the corresponding counter
%
\newcounter{subsectiontmp}
\newcounter{subsubsectiontmp}
\newcommand{\sectionlabel}{%
\setcounter{subsectiontmp}{\value{subsection}}%
\setcounter{subsubsectiontmp}{\value{subsubsection}}%
\ifnum\value{section}=0%
{}%
\else%
\thesection%
\fi%
\ifnum\value{subsectiontmp}=0%
{}%
\else%
{.\thesubsectiontmp}%
\fi%
\ifnum\value{subsubsectiontmp}=0%
{}%
\else%
{.\thesubsubsectiontmp}%
\fi%
}

%
% this command evaluates to the title, given the title name, it is rotated depending on the counter flag
%
% #1 - title
%
\newcommand{\keycardtitle}[1]{%
	\ifnum\value{printVersion}=1%
		\hspace*{\fill}\rotatebox[origin=c]{180}{\parbox[t][][t]{0.48\textwidth}{\textbf{\sectionlabel\ #1}}}%
	\else%
		\textbf{\sectionlabel\ #1}%
	\fi%
}

%
% Parbox with approximately 0.5 textwidth and 0.25 text height that displays the given title and content
%
% #1 - title of the key card, i.e., the reference of the keycard's content
% #2 - content of the key card
%
\newcommand{\keycard}[2]{%
\parbox[t][0.248\textheight][t]{0.48\textwidth}{%
	\keycardtitle{#1}
	
	\vspace{0.5cm}%
	#2% 
}%
}

%
% Behaves similar as \keycard with the difference that the given content may span the whole text width and also multiple pages
%
\newcommand{\longkeycard}[2]{%
\vspace{0.5cm}%
\hrule%
\vspace{0.5cm}%
\keycardtitle{#1}%
		
\vspace{0.5cm}%
#2%
\vspace{0.5cm}%
\hrule%
\vspace{0.5cm}%
}
