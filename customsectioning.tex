%
% Registers a section (to be used in the table of contents and some other places), but doesn't print it
% #1 - name of the section
%
\newcommand{\ksection}[1]{%
\phantomsection%
\stepcounter{section}%
\addcontentsline{toc}{section}{\protect\numberline{\thesection}#1}%
}

%
% Analogous to section command, but with subsection
%
\newcommand{\ksubsection}[1]{%
\phantomsection%
\stepcounter{subsection}%
\addcontentsline{toc}{subsection}{\protect\numberline{\thesubsection}#1}%
}

%
% Analogous to section command, but with subsubsection
%
\newcommand{\ksubsubsection}[1]{%
\phantomsection%
\stepcounter{subsubsection}%
\addcontentsline{toc}{subsubsection}{\protect\numberline{\thesubsubsection}#1}%
}
